    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>爱心祝福弹窗</title>
    <style>
        body {
            margin: 0;
            padding: 0;
            background-color: #000;
            overflow: hidden;
        }
        .popup {
            position: absolute;
            background-color: #fff;
            padding: 10px;
            border-radius: 5px;
            box-shadow: 0 0 10px rgba(0,0,0,0.2);
            font-family: "Microsoft YaHei", sans-serif;
            text-align: center;
            opacity: 0.9;
            animation: fadeIn 0.5s ease-in-out;
        }
        @keyframes fadeIn {
            from { opacity: 0; transform: scale(0.5); }
            to { opacity: 0.9; transform: scale(1); }
        }
    </style>

<body>
    <button onclick="showInfo()">点击开始</button>
    <script>
        const messages = [
            "保持微笑", "多喝水", "天冷了，多穿衣服", "愿所有烦恼都消失", "保持好心情",
            "期待下一次见面", "我想你了", "顺顺利利", "别熬夜", "今天过得开心嘛",
            "好好爱自己", "早点休息", "记得吃水果", "多喝温水~", "照顾好自己"
        ];
        const colors = [
            '#fceabb', '#f8b500', '#f4a460', '#e9967a', '#dda0dd', '#ba55d3', 
            '#9370db', '#87ceeb', '#add8e6', '#b0e0e6', '#90ee90', '#98fb98', 
            '#f0f8ff', '#e6e6fa'
        ];
        // 适配手机的爱心坐标
        const heartCoords = [
            [30, 10], [80, 30], [130, 10],
            [0, 60], [50, 80], [100, 60], [150, 60],
            [-50, 110], [0, 130], [50, 110], [100, 130], [150, 110], [200, 110],
            [-100, 160], [-50, 180], [0, 160], [50, 180], [100, 160], [150, 180], [200, 160], [250, 160],
            [-150, 210], [-100, 230], [-50, 210], [0, 230], [50, 210], [100, 230], [150, 210], [200, 230], [250, 210], [300, 210],
            [-200, 260], [-150, 280], [-100, 260], [-50, 280], [0, 260], [50, 280], [100, 260], [150, 280], [200, 260], [250, 280], [300, 260], [350, 260],
            [-250, 310], [-200, 330], [-150, 310], [-100, 330], [-50, 310], [0, 330], [50, 310], [100, 330], [150, 310], [200, 330], [250, 310], [300, 330], [350, 310], [400, 310],
            [-200, 360], [-150, 380], [-100, 360], [-50, 380], [0, 360], [50, 380], [100, 360], [150, 380], [200, 360], [250, 380], [300, 360], [350, 380],
            [-150, 410], [-100, 430], [-50, 410], [0, 430], [50, 410], [100, 430], [150, 410], [200, 430], [250, 410], [300, 430],
            [-100, 460], [-50, 480], [0, 460], [50, 480], [100, 460], [150, 480], [200, 460],
            [-50, 510], [0, 530], [50, 510], [100, 530], [150, 510],
            [0, 560], [50, 580], [100, 560],
            [50, 610]
        ];

        function createPopup() {
            const popup = document.createElement('div');
            popup.className = 'popup';
            const message = messages[Math.floor(Math.random() * messages.length)];
            const color = colors[Math.floor(Math.random() * colors.length)];
            const [x, y] = heartCoords[Math.floor(Math.random() * heartCoords.length)];
            // 适配手机屏幕居中
            popup.style.left = `${x + window.innerWidth/2 - 75}px`;
            popup.style.top = `${y + window.innerHeight/4}px`;
            popup.style.backgroundColor = color;
            popup.style.width = '130px';
            popup.style.height = '60px';
            popup.innerHTML = `<div style="margin-top: 15px;">${message}</div>`;
            document.body.appendChild(popup);

            // 3秒后淡出并移除
            setTimeout(() => {
                popup.style.opacity = '0';
                popup.style.transition = 'opacity 0.5s ease-in-out';
                setTimeout(() => {
                    document.body.removeChild(popup);
                }, 500);
            }, 3000);
        }

        function showInfo() {
            const info = document.createElement('div');
            info.style.position = 'absolute';
            info.style.left = '50%';
            info.style.top = '50%';
            info.style.transform = 'translate(-50%, -50%)';
            info.style.backgroundColor = '#fff';
            info.style.padding = '20px';
            info.style.borderRadius = '5px';
            info.style.boxShadow = '0 0 10px rgba(0,0,0,0.2)';
            info.innerHTML = `
                <div style="font-size: 18px; font-weight: bold; text-align: center; margin-bottom: 10px;">祝你天天开心</div>
                <button onclick="startPopups()">好</button>
            `;
            document.body.appendChild(info);
        }

        function startPopups() {
            const info = document.querySelector('div[style*="translate(-50%, -50%)"]');
            document.body.removeChild(info);
            // 批量生成弹窗
            for (let i = 0; i < 50; i++) {
                setTimeout(createPopup, i * 100);
            }
        }
    </script>
</body>
